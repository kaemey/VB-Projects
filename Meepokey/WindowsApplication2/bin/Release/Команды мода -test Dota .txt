-gold <amount> // выдать указанное кол-во золота

-time <00-24> // выставить время суток

-spawncreeps // заспаунить крипов на лайнах (не работает вместе с -spawnoff)

-powerup // создать руны

-killall // убить сентов и скоржей

-killsent // убить сентов

-killscourge // убить скоржей

-neutrals // заспаунить нейтралов

-noherolimit // позволить выбрать одному игроку нескольких героев в таверне

-trees // пересоздать деревья

-killwards // бить все варды

-spawnoff/-spawnon // выключить/включить спаун крипов на лайнах по таймеру

-roshan // пересоздать рошана

-respawn // возродить основного героя (основной - тот, который отображен в таблице результатов)

-dummy // создать дамми юнита

-kill (-k) // убить выделенных юнитов

-remove // удалить с карты выделенных юнитов

-refresh(-r) // сбросить состояние выделенных юнитов (восстановить ХП/МП/Кулдауны)

-lvl (-l) <1-25> // добавить указанное количество уровней выделенным юнитам

-unit (-u) <ally(a)/enemy(e)> <name/id>[x2/x3/x4...] // создать указанное количество союзных/вражеских юнитов по имени или айди

-abils // перечислить абилки выделенных юнитов

-items // перечислить предметы выделенных юнитов

-abil <code> <lvl> // выдать юниту абилку требуемого уровня (0й уровень - удалить)

-int <count> // добавить юниту указанное кол-во интеллекта

-str <count> // добавить юниту указанное кол-во силы

-agi <count> // добавить юниту указанное кол-во ловкости

-item (-i) <name/id> [x2/x3/x4...] // выдать указанное количество предметов выделенным юнитам по имени или айди

-fullcontrol (-fc) // включить/выключить полный контроль над юнитами союзников и врагов

-damage (-dmg) // включить/выключить наблюдение за уроном выделенных юнитов

-dealdamage (-dd) <amount> // нанести указанное кол-во физического урона выделенным юнитам

-owner (-o) // сменить владельца юнита (1-5 сенты, 6-10 - скоржи)

-vertex <RR> <GG> <BB> <AA> // наложить цвет на выделенных юнитов

-anim <index> // проиграть анимацию юнита по индексу

-ums // вывести скорость передвижения выделленных юнитов

-autoattack (-aa) // включить/выключить автоатаку на карте

-santa // дать шапку санты выделенным юнитам

-builder // включить режим "строитель", позволяющий перемещать выделенные здания с помощью стрелок на клавиатуре

-distance // показать дистанцию между двумя выделенными юнитами

-timings // включить/выключить наблюдение за таймингами каста абилок выделенных юнитов. Отображает временные характеристики каждого состояния каста абилки

Могу помочь с тестами.
